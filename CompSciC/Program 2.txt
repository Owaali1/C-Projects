#include <iostream>
#include <iomanip>
using namespace std;
int main(void)
{
	float a, b, c;
	cout << "Enter 3 real numbers ";
	cin >> a >> b >> c;
	cout << fixed << setprecision(2);
	//simple ifs
	if (a > 0 && b > 0 && c > 0) cout << "Sum:" << setw(12) << a + b + c << endl;
	//nested ifs 
	if (a > 0) if (b > 0) if (c > 0) cout << "All numbers are positive" << endl;
	else cout << "First number is positive, second is positive, third is not" << " Product:" << setw(12) << a * b << endl;
	else cout << "First number is positive, second is not, third is positive" << " Product:" << setw(12) << a * c << endl;
	else cout << "First number is not, second is positive, third is positive" << " Product:" << setw(12) << b * c << endl;
	float x, y;
	cout << "Enter 2 real numbers ";
	cin >> x >> y;
	cout << fixed << setprecision(2);
	//if-else
	if (x < 0 && y < 0) cout << "Both numbers are negative" << " Quotient:" << setw(12) << x / y << endl;
	else cout << "Both numbers have to be negative!" << endl;
	return 0;


}
